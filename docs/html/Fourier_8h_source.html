103 template<typename RT, typename IOT>
104 void Fourier::powerDump(RT *in, IOT *iodata)
105 {
106  // Transform input array
107 
108  fftw_execute_dft_r2c(p_r2c, in, f_field);
109 
110  // average power over angles
111  const int numbins = (int) (sqrt(NX*NX + NY*NY + NZ*NZ)/2.0) + 1; // Actual number of bins
112  RT * array_out = new RT[numbins];
113  int * numpoints = new int[numbins]; // Number of points in each momentum bin
114  RT * p = new RT[numbins];
115  RT * f2 = new RT[numbins]; // Values for each bin: Momentum, |F-k|^2, n_k
116 
117  double pmagnitude; // Total momentum (p) in units of lattice spacing, pmagnitude = Sqrt(px^2+py^2+pz^2).
118  // This also gives the bin index since bin spacing is set to equal lattice spacing.
119  double fp2;
120  int i, j, k, px, py, pz; // px, py, and pz are components of momentum in units of grid spacing
121 
122  // Initial magnitude of momentum in each bin
123  for(i=0; i<numbins; i++) {
124  f2[i] = 0.0;
125  numpoints[i] = 0;
126  }
127 
128  // Perform average over all angles here (~integral d\Omega).
129  for(i=0; i<NX; i++)
130  {
131  px = (i<=NX/2 ? i : i-NX);
132  for(j=0; j<NY; j++)
133  {
134  py = (j<=NY/2 ? j : j-NY);
135  for(k=1; k<NZ/2; k++)
136  {
137  pz = k;
138  pmagnitude = sqrt((RT) (pw2(px) + pw2(py) + pw2(pz)));
139  fp2 = pw2(C_RE((f_field)[FFT_NP_INDEX(i,j,k)])) + pw2(C_IM((f_field)[FFT_NP_INDEX(i,j,k)]));
140  numpoints[(int)pmagnitude] += 2;
141  f2[(int)pmagnitude] += 2.*fp2;
142  }
143 
144  pz = 0;
145  k = 0;
146  pmagnitude = sqrt((RT) (pw2(px) + pw2(py) + pw2( pz)));
147  fp2 = pw2(C_RE((f_field)[FFT_NP_INDEX(i,j,k)])) + pw2(C_IM((f_field)[FFT_NP_INDEX(i,j,k)]));
148  numpoints[(int)pmagnitude] += 1;
149  f2[(int)pmagnitude] += fp2;
150 
151  pz = NZ/2;
152  k = NZ/2;
153  pmagnitude = sqrt((RT) (pw2(px) + pw2(py) + pw2(pz)));
154  fp2 = pw2(C_RE((f_field)[FFT_NP_INDEX(i,j,k)])) + pw2(C_IM((f_field)[FFT_NP_INDEX(i,j,k)]));
155  numpoints[(int)pmagnitude] += 1;
156  f2[(int)pmagnitude] += fp2;
157  }
158  }
159 
160  for(i=0; i<numbins; i++)
161  {
162  // Converts sums to averages. (numpoints[i] should always be greater than zero.)
163  if(numpoints[i] > 0)
164  {
165  array_out[i] = f2[i]/((double) numpoints[i]);
166  }
167  else
168  {
169  array_out[i] = 0.;
170  }
171  }
172 
173  // write data
174  std::string filename = iodata->dir() + "spec.dat.gz";
175  char data[20];
176 
177  gzFile datafile = gzopen(filename.c_str(), "ab");
178  if(datafile == Z_NULL) {
179  printf("Error opening file: %s\n", filename.c_str());
180  return;
181  }
182 
183  for(i=0; i<numbins; i++)
184  {
185  // field values
186  sprintf(data, "%g\t", (double) array_out[i]);
187  gzwrite(datafile, data, std::char_traits<char>::length(data));
188  }
189  gzwrite(datafile, "\n", std::char_traits<char>::length("\n"));
190 
191  gzclose(datafile);
192  delete[] array_out;
193  delete[] numpoints;
194  delete[] p;
195  delete[] f2;
196 
197  return;
198 }

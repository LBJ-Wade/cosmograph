155  void setShiftFn(std::string name)
156  {
157  // Shift needs to be enabled for non-trivial evolution
158  if(name != "Static" && !USE_BSSN_SHIFT)
159  {
160  std::cerr << "Code must be compiled with shift enabled to use non-Static shift!";
161  throw -1;
162  }
163 
164  // Gamma driver needs an extra field compiled in
165  if(name == "gammadriver" && !USE_GAMMA_DRIVER)
166  {
167  std::cerr << "Code must be compiled with gamma driver enabled to use the gamma driver gauge.";
168  throw -1;
169  }
170 
171  if ( shift_gauge_map.find(name) == shift_gauge_map.end() )
172  {

174  throw -1;
175  }
176 
177  shift_fn1 = shift_gauge_map[name]["1"];

179  shift_fn3 = shift_gauge_map[name]["3"];

164  inline REAL_T _laplacian(fas_grid_t grid, IDX_T i, IDX_T j, IDX_T k, IDX_T nx, IDX_T ny, IDX_T nz)
165  {
166  REAL_T dx = grid_length_x/nx, dy = grid_length_y/ny, dz = grid_length_z/nz;
167 
168  return (
169  - 1.0/560.0*grid[_gIdx(i-4,j,k,nx,ny,nz)]
170  + 8.0/315.0*grid[_gIdx(i-3,j,k,nx,ny,nz)]
171  - 1.0/5.0*grid[_gIdx(i-2,j,k,nx,ny,nz)]
172  + 8.0/5.0*grid[_gIdx(i-1,j,k,nx,ny,nz)]
173  - 205.0/72.0*grid[_gIdx(i-0,j,k,nx,ny,nz)]
174  + 8.0/5.0*grid[_gIdx(i+1,j,k,nx,ny,nz)]
175  - 1.0/5.0*grid[_gIdx(i+2,j,k,nx,ny,nz)]
176  + 8.0/315.0*grid[_gIdx(i+3,j,k,nx,ny,nz)]
177  - 1.0/560.0*grid[_gIdx(i+4,j,k,nx,ny,nz)]
178  )/dx/dx + (
179  - 1.0/560.0*grid[_gIdx(i,j-4,k,nx,ny,nz)]
180  + 8.0/315.0*grid[_gIdx(i,j-3,k,nx,ny,nz)]
181  - 1.0/5.0*grid[_gIdx(i,j-2,k,nx,ny,nz)]
182  + 8.0/5.0*grid[_gIdx(i,j-1,k,nx,ny,nz)]
183  - 205.0/72.0*grid[_gIdx(i,j-0,k,nx,ny,nz)]
184  + 8.0/5.0*grid[_gIdx(i,j+1,k,nx,ny,nz)]
185  - 1.0/5.0*grid[_gIdx(i,j+2,k,nx,ny,nz)]
186  + 8.0/315.0*grid[_gIdx(i,j+3,k,nx,ny,nz)]
187  - 1.0/560.0*grid[_gIdx(i,j+4,k,nx,ny,nz)]
188  )/dy/dy + (
189  - 1.0/560.0*grid[_gIdx(i,j,k-4,nx,ny,nz)]
190  + 8.0/315.0*grid[_gIdx(i,j,k-3,nx,ny,nz)]
191  - 1.0/5.0*grid[_gIdx(i,j,k-2,nx,ny,nz)]
192  + 8.0/5.0*grid[_gIdx(i,j,k-1,nx,ny,nz)]
193  - 205.0/72.0*grid[_gIdx(i,j,k-0,nx,ny,nz)]
194  + 8.0/5.0*grid[_gIdx(i,j,k+1,nx,ny,nz)]
195  - 1.0/5.0*grid[_gIdx(i,j,k+2,nx,ny,nz)]
196  + 8.0/315.0*grid[_gIdx(i,j,k+3,nx,ny,nz)]
197  - 1.0/560.0*grid[_gIdx(i,j,k+4,nx,ny,nz)]
198  )/dz/dz;
199  }

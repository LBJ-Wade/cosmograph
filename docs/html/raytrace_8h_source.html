279  void initializeScreenVectors()
280  {
281  // S_A is spatial, _|_ V
282  RT s1b[3] = {0.0, 1.0, 0.0}; // basis vector for S1
283  RT s2b[3] = {1.0, 0.0, 0.0}; // basis vector for S2
284 
285  // special case if in x- or y-direction; or x-y plane
286  if(std::abs(rd.V[2]) <= 1e-10 && std::abs(rd.V[1]) <= 1e-10)
287  {
288  s2b[2] = 1.0;
289  s2b[0] = 0.0;
290  }
291  else if(std::abs(rd.V[2]) <= 1e-10 && std::abs(rd.V[0]) <= 1e-10)
292  {
293  s1b[2] = 1.0;
294  s1b[1] = 0.0;
295  }
296  else
297  {
298  // if V is in x-y plane, pick vectors out of that plane
299  if(std::abs(rd.V[2]) <= 1e-10) { s1b[2] = 1.0; s2b[2] = 1.0; }
300  // if V is in x-z plane, pick vectors out of that plane
301  if(std::abs(rd.V[1]) <= 1e-10) { s2b[1] = 1.0; }
302  // if V is in y-z plane, pick vectors out of that plane
303  if(std::abs(rd.V[0]) <= 1e-10) { s1b[0] = 1.0; }
304  }
305 
306  // store result in S1, S2
307  for(int i=0; i<3; i++)
308  {
309  rd.S1[i] = s1b[i];
310  rd.S2[i] = s2b[i];
311  }
312 
313  return;
314  }

316  void orthonormalizeScreenVectors()
317  {
318  RT new_S1[3], new_S2[3];
319  RT s1b[3] = {rd.S1[0], rd.S1[1], rd.S1[2]}; // temporary basis vector for S1
320  RT s2b[3] = {rd.S2[0], rd.S2[1], rd.S2[2]}; // temporary basis vector for S2
321 
322  // subtract out part of s1 along V & normalize
323  for(int i=1; i<=3; i++)
324  {
325  new_S1[iIDX(i)] = s1b[iIDX(i)] - (
326  rp.g[aIDX(1,1)]*s1b[iIDX(1)]*rd.V[iIDX(1)] + rp.g[aIDX(2,2)]*s1b[iIDX(2)]*rd.V[iIDX(2)] + rp.g[aIDX(3,3)]*s1b[iIDX(3)]*rd.V[iIDX(3)]
327  + rp.g[aIDX(1,2)]*s1b[iIDX(1)]*rd.V[iIDX(2)] + rp.g[aIDX(1,3)]*s1b[iIDX(1)]*rd.V[iIDX(3)] + rp.g[aIDX(2,3)]*s1b[iIDX(2)]*rd.V[iIDX(3)]
328  + rp.g[aIDX(1,2)]*s1b[iIDX(2)]*rd.V[iIDX(1)] + rp.g[aIDX(1,3)]*s1b[iIDX(3)]*rd.V[iIDX(1)] + rp.g[aIDX(2,3)]*s1b[iIDX(3)]*rd.V[iIDX(2)]
329  )*rd.V[iIDX(i)];
330  }
331  RT mags1 = std::sqrt(
332  rp.g[aIDX(1,1)]*new_S1[iIDX(1)]*new_S1[iIDX(1)] + rp.g[aIDX(2,2)]*new_S1[iIDX(2)]*new_S1[iIDX(2)] + rp.g[aIDX(3,3)]*new_S1[iIDX(3)]*new_S1[iIDX(3)]
333  + 2.0*(rp.g[aIDX(1,2)]*new_S1[iIDX(1)]*new_S1[iIDX(2)] + rp.g[aIDX(1,3)]*new_S1[iIDX(1)]*new_S1[iIDX(3)] + rp.g[aIDX(2,3)]*new_S1[iIDX(2)]*new_S1[iIDX(3)])
334  );
335  for(int i=1; i<=3; i++)
336  {
337  new_S1[iIDX(i)] /= mags1;
338  }
339 
340  // subtract out part of s2 along V & s1
341  for(int i=1; i<=3; i++)
342  {
343  new_S2[iIDX(i)] = s2b[iIDX(i)] - (
344  rp.g[aIDX(1,1)]*s2b[iIDX(1)]*rd.V[iIDX(1)] + rp.g[aIDX(2,2)]*s2b[iIDX(2)]*rd.V[iIDX(2)] + rp.g[aIDX(3,3)]*s2b[iIDX(3)]*rd.V[iIDX(3)]
345  + 2.0*(rp.g[aIDX(1,2)]*s2b[iIDX(1)]*rd.V[iIDX(2)] + rp.g[aIDX(1,3)]*s2b[iIDX(1)]*rd.V[iIDX(3)] + rp.g[aIDX(2,3)]*s2b[iIDX(2)]*rd.V[iIDX(3)])
346  )*rd.V[iIDX(i)] - (
347  rp.g[aIDX(1,1)]*s2b[iIDX(1)]*new_S1[iIDX(1)] + rp.g[aIDX(2,2)]*s2b[iIDX(2)]*new_S1[iIDX(2)] + rp.g[aIDX(3,3)]*s2b[iIDX(3)]*new_S1[iIDX(3)]
348  + 2.0*(rp.g[aIDX(1,2)]*s2b[iIDX(1)]*new_S1[iIDX(2)] + rp.g[aIDX(1,3)]*s2b[iIDX(1)]*new_S1[iIDX(3)] + rp.g[aIDX(2,3)]*s2b[iIDX(2)]*new_S1[iIDX(3)])
349  )*new_S1[iIDX(i)];
350  }
351  RT mags2 = std::sqrt(
352  rp.g[aIDX(1,1)]*new_S2[iIDX(1)]*new_S2[iIDX(1)] + rp.g[aIDX(2,2)]*new_S2[iIDX(2)]*new_S2[iIDX(2)] + rp.g[aIDX(3,3)]*new_S2[iIDX(3)]*new_S2[iIDX(3)]
353  + 2.0*(rp.g[aIDX(1,2)]*new_S2[iIDX(1)]*new_S2[iIDX(2)] + rp.g[aIDX(1,3)]*new_S2[iIDX(1)]*new_S2[iIDX(3)] + rp.g[aIDX(2,3)]*new_S2[iIDX(2)]*new_S2[iIDX(3)])
354  );
355  for(int i=1; i<=3; i++)
356  {
357  new_S2[iIDX(i)] /= mags2;
358  }
359 
360  // store result in S1, S2
361  for(int i=0; i<3; i++)
362  {
363  rd.S1[i] = new_S1[i];
364  rd.S2[i] = new_S2[i];
365  }
366 
367  return;
368  }

383  );

387  {

396  *( 2.0*rp.gi[aIDX(i,3)] - rd.V[iIDX(i)]*rd.V[iIDX(3)] )
397  - rp.G[iIDX(i)][aIDX(1,1)]*rd.V[iIDX(1)]*rd.V[iIDX(1)]
398  - rp.G[iIDX(i)][aIDX(1,2)]*rd.V[iIDX(1)]*rd.V[iIDX(2)]
399  - rp.G[iIDX(i)][aIDX(1,3)]*rd.V[iIDX(1)]*rd.V[iIDX(3)]

402  - rp.G[iIDX(i)][aIDX(2,3)]*rd.V[iIDX(2)]*rd.V[iIDX(3)]
403  - rp.G[iIDX(i)][aIDX(3,1)]*rd.V[iIDX(3)]*rd.V[iIDX(1)]

405  - rp.G[iIDX(i)][aIDX(3,3)]*rd.V[iIDX(3)]*rd.V[iIDX(3)]
406  );

408  // lowered velocity vectors

411  RT V_3 = rp.g[aIDX(3,1)]*rd.V[iIDX(1)] + rp.g[aIDX(3,2)]*rd.V[iIDX(2)] + rp.g[aIDX(3,3)]*rd.V[iIDX(3)];
412  // V_j V^k \Gamma^j_{ki} for a free index i

415  + V_1*rp.G[iIDX(1)][aIDX(2,1)]*rd.V[iIDX(2)] + V_2*rp.G[iIDX(2)][aIDX(2,1)]*rd.V[iIDX(2)] + V_3*rp.G[iIDX(3)][aIDX(2,1)]*rd.V[iIDX(2)]

417  );
